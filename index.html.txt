    <div class="panel">
        <b style="font-size: 14px;">🗺️ Добавление поля</b>
        <input type="text" id="fieldName" placeholder="Название (напр. Поле №1)">
        <button onclick="saveField()">💾 Сохранить поле</button>
        <button class="clear-btn" onclick="location.reload()">❌ Сбросить точки</button>
    </div>

    <div id="map"></div>

    <script>
        const tg = window.Telegram.WebApp;
        tg.expand(); 

        // Центр возле Никополя
        const map = L.map('map').setView([47.5746, 34.4396], 13);
        
        // Исправленная карта (OpenStreetMap)
        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
            attribution: '&copy; OpenStreetMap'
        }).addTo(map);

        let activePoints = [];
        let currentPolygon = null;

        map.on('click', function(e) {
            const lat = e.latlng.lat;
            const lng = e.latlng.lng;
            activePoints.push([lat, lng]);

            if (currentPolygon) map.removeLayer(currentPolygon);
            if (activePoints.length >= 2) {
                currentPolygon = L.polygon(activePoints, {color: '#28a745'}).addTo(map);
            } else {
                L.marker([lat, lng]).addTo(map);
            }
        });

        function saveField() {
            const name = document.getElementById('fieldName').value.trim();
            if (!name || activePoints.length < 3) { alert('Введите название и отметьте точки!'); return; }
            tg.sendData(JSON.stringify({name: name, points: activePoints}));
            tg.close();
        }
    </script>
