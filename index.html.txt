    <div class="panel">
        <b>🗺️ Мои Поля</b>
        <input type="text" id="fieldName" placeholder="Название поля">
        <button onclick="saveField()">💾 Сохранить</button>
        <button style="background:#dc3545;" onclick="location.reload()">❌ Сброс</button>
    </div>
    <div id="map"></div>
    <script>
        const tg = window.Telegram.WebApp;
        tg.expand();
        
        // Карта со спутника (Esri World Imagery)
        const map = L.map('map').setView([47.5746, 34.4396], 15);
        L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}', {
            attribution: 'Tiles &copy; Esri &mdash; Source: Esri, i-cubed, USDA, USGS, AEX, GeoEye, Getmapping, Aerogrid, IGN, IGP, UPR-EGP, and the GIS User Community'
        }).addTo(map);

        let activePoints = [];
        let currentPolygon = null;

        map.on('click', function(e) {
            activePoints.push([e.latlng.lat, e.latlng.lng]);
            if (currentPolygon) map.removeLayer(currentPolygon);
            currentPolygon = L.polygon(activePoints, {color: '#ffff00', weight: 3}).addTo(map);
        });

        function saveField() {
            const name = document.getElementById('fieldName').value.trim();
            if (!name || activePoints.length < 3) { alert('Введите название и кликните точки!'); return; }
            tg.sendData(JSON.stringify({name: name, points: activePoints}));
            tg.close();
        }
    </script>
